Code Sample: Drawing a Cube

// Set up 3D view
glTranslatef(0.0, 0.0, -5.0);
glRotatef(30, 1.f, 1.f, 0.f);

// Draw cube faces with colors
glBegin(GL_QUADS);
  // Front face - Green
  glColor3f(0.0f, 1.0f, 0.0f);
  glVertex3f(-0.5f, -0.5f, 0.5f);
  glVertex3f( 0.5f, -0.5f, 0.5f);
  glVertex3f( 0.5f,  0.5f, 0.5f);
  glVertex3f(-0.5f,  0.5f, 0.5f);
  // ... more faces
glEnd();
